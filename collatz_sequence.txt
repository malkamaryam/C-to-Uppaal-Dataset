using System;

class CollatzSequence
{
    static void Main()
    {
        int n = 7;
        int steps = 0;

        while (n != 1)
        {
            if (n % 2 == 0)
                n = n / 2;
            else
                n = 3 * n + 1;

            steps++;
        }

        int result = steps;
        Console.WriteLine(result);
    }
}
