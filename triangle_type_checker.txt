using System;

class TriangleType
{
    static void Main()
    {
        int a = 5, b = 5, c = 5;
        int type;

        if (a + b <= c || a + c <= b || b + c <= a)
            type = 0;
        else if (a == b && b == c)
            type = 1;
        else if (a == b || b == c || a == c)
            type = 2;
        else
            type = 3;

        Console.WriteLine(type);
    }
}
