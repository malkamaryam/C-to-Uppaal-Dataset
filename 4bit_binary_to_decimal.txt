using System;

class BinaryToDecimal
{
    static void Main()
    {
        int binary = 1011;  // 4-bit binary number
        int decimalValue = 0;

        // Compute decimal value directly
        decimalValue = (binary % 10) 
                     + ((binary / 10) % 10) * 2
                     + ((binary / 100) % 10) * 4
                     + ((binary / 1000) % 10) * 8;

        Console.WriteLine("Decimal value: " + decimalValue);
    }
}
