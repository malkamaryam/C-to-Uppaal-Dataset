using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        List<int> numbers = new List<int>{5, 3, 12, 9, 1};
        int max = numbers[0];

        foreach(int n in numbers)
        {
            if(n > max)
                max = n;
        }

        Console.WriteLine("Max: " + max);
    }
}
