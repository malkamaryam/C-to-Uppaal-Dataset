using System;

class ProductOfDigits
{
    static void Main()
    {
        int num = 19;   // fixed input
        int product = 1;

        while (num > 0)
        {
            int digit = num % 10;
            product *= digit;
            num /= 10;
        }

        int result = product;
        Console.WriteLine(result);
    }
}
