using System;

class GradeCalculator
{
    static void Main()
    {
        int marks = 85;
        int grade;

        if (marks >= 90)
            grade = 1;
        else if (marks >= 80)
            grade = 2;
        else if (marks >= 70)
            grade = 3;
        else if (marks >= 60)
            grade = 4;
        else
            grade = 5;

        Console.WriteLine(grade);
    }
}
